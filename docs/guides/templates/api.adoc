  <#return text?replace("{", "\\{")>
</#function>

</#if>
----

<#if (op.parameters)?has_content>
<#assign pathParams = op.parameters?filter(p -> (p.in)! == "path")>

|===
|Name |Description

|===

</#if>
<#if queryParams?has_content>
